Con \(\prec\) tal que devuelve cualquier lista diferente de \(\bot\)

f solve(v,w):

  memo = matriz de n*w

  f os(i,s):
    
    si (i = n and s = w) or (s > w):
      return None

    si i = n and i = w:
      return []
      
    si memo[i][s] != None:
      return memo[i][s]
 
    //prec retorna cualquiera de los 3 si es distinto de None
    memo[i][s] = prec(os(i+1,s+v[i+1])++[+],
                      os(i+1,s*v[i+1])++[x],
                      os(i+1,s^v[i+1])++[↑])
    
    return memo[i][s]
